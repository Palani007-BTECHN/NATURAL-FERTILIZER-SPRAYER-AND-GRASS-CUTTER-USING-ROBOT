void loop()
{ 
 
  while(1)
  {
    if(Serial1.available()>0)
    {
      rw=Serial1.read();
     Serial.println(rw);
      if(rw=='A')
      {
         
        Serial1.println("auto mode");
        
        automode();
      
      }
      else if(rw=='M')
      {
          Serial1.println("manual mode");
         
        manualmode();
      }
   }
 }
}
void automode()
{
  while(1)
  {
  digitalWrite(trigger,LOW);
  delayMicroseconds(5);
  digitalWrite(trigger,HIGH);
  delayMicroseconds(10);
  digitalWrite(trigger,HIGH);
  int distance=pulseIn(echo_pin,HIGH);
  int dist=distance*0.034/2;
  Serial1.print("Dist: ");
  Serial1.println(dist);
  
  
  delay(800);
  if(dist<30)
  {
    Serial1.println("Object Detected");
    
    stop_robot();
   delay(1000); 
  }
  else
  {

digitalWrite(grass1,HIGH);
digitalWrite(grass2,LOW);
delay(1500);
  forward_movement();
 // Grass_cutter();
  delay(2000);

  digitalWrite(robot1,LOW);
   digitalWrite(robot2,LOW);
   digitalWrite(robot3,LOW);
   digitalWrite(robot4,LOW);
  delay(1000); 
  
  // seed_dropon();
  // delay(1000);

  forward_movement();
  //Grass_cutter();
  delay(2000);

   
   digitalWrite(robot1,LOW);
   digitalWrite(robot2,LOW);
   digitalWrite(robot3,LOW);
   digitalWrite(robot4,LOW);
  delay(1000);
   
 
  Moisture_Read();
  
  if(M_val<200)
  {
    
    
    Serial1.println("SOIL IS DRY");
    digitalWrite(relay1,HIGH);
    delay(2000);
    digitalWrite(relay1,LOW);
    delay(1000);
  }
  else
  {
   Serial1.println("SOIL IS WET");
   
   digitalWrite(relay1,LOW); 
   delay(500);
  }
  temp_check(); 
  Pesticide_Check(); 
  

}

}
}

void manualmode()
  {
    while(1)
    {
     if(Serial1.available() > 0)
      {

        ch = Serial1.read();
          
        Serial.print(ch);
            if(ch=='A')
            {
              
              Serial1.println("auto mode");
              automode();
            }
           else if(ch=='F')
            {
              forward_movement();
             
            
            }
          else if(ch=='B')
            {
              
              back_movement();
            }
          else if(ch=='L')
            {
              
              left_movement();
             }
          else if(ch=='R')
            {
              
              right_movement();
            }
                    
           else if(ch=='S')
            {
              
               stop_robot();
            }
           
           else if(ch=='G')
            {
              Grass_cutter();
              
              delay(500);
              
            }

            else if(ch=='H')
            {
              Grass_cutter_off();
             
              delay(500);
              
            }
           
            else if(ch=='T')
            {
              temp_check();
              
              
            }
           else if(ch=='P')
            {
              
              Serial1.println("Pesticide Spraying");
              digitalWrite(relay1,HIGH);
              delay(2000);
              digitalWrite(relay1,LOW);
              delay(1000);
              
           }
           else if(ch=='W')
            {
              
              
              Serial1.println("Water Pump On");
            
              digitalWrite(relay2,HIGH);
              delay(2000);
              digitalWrite(relay2,LOW);
              delay(1000);
           }
          else if(ch=='M')
            {
              Moisture_Read();
              if(M_val<200)
              {
               
                Serial1.println("SOIL IS DRY");
                
                delay(1000);
                digitalWrite(relay2,HIGH);
                delay(1000);
              }
              else
              {
               Serial1.println("SOIL IS WET");
               
               delay(1000);
               digitalWrite(relay2,LOW); 
              }
              
           }
        
     }
     Pesticide_Check();
     
  }

}

void Pesticide_Check()
{

  if(Serial.available()>0)
  {
    char chr=Serial.read();
    Serial1.println(chr);
    delay(500);

    if(chr=='A')
    {
  
  Serial1.println("Pesticide Spraying");
 
  digitalWrite(relay1,HIGH);
  delay(2000);
  digitalWrite(relay1,LOW);
  delay(1000);
    }

    if(chr=='B')
    {
    
    Serial1.println("Healthy Plant");
    
    delay(2000);
    }
  }
 


  
  
  
}
void Moisture_Read()
{
  M_val=analogRead(A2);
  M_val=1023-M_val;
  
   
  
  Serial1.println("Moisture:"+String(M_val));
   
  delay(1000);
}
void temp_check()
{
  DHT.read11(dht_dpin);
      Humidity=DHT.humidity;
      tempc=DHT.temperature;
      Serial1.print("$Humidity: ");
      Serial1.print(DHT.humidity);   // printing Humidity on LCD
      Serial1.print(" %");
      Serial1.print(" # ");
  
      Serial1.print("$Temperature:");
      Serial1.print(DHT.temperature);   // Printing temperature on LCD
      Serial1.print(" C");
      Serial1.println(" # ");

 

     
     
     delay(1000);
}
void forward_movement()
{
  Serial1.println("FORWARD");
  

   digitalWrite(robot1,HIGH);
   digitalWrite(robot2,LOW);
   digitalWrite(robot3,HIGH);
   digitalWrite(robot4,LOW);
}
void back_movement()
{
  Serial1.println("REVERSE");

   digitalWrite(robot1,LOW);
   digitalWrite(robot2,HIGH);
   digitalWrite(robot3,LOW);
   digitalWrite(robot4,HIGH);
}

void left_movement()
{
  Serial1.println("LEFT");
  digitalWrite(robot1,HIGH);
   digitalWrite(robot2,LOW);
   digitalWrite(robot3,LOW);
   digitalWrite(robot4,HIGH);
 
   
}
void right_movement()
{  
 Serial1.println("RIGHT");
 digitalWrite(robot1,LOW);
   digitalWrite(robot2,HIGH);
   digitalWrite(robot3,HIGH);
   digitalWrite(robot4,LOW);
 
   
}

{

}
